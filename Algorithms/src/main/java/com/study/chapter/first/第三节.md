## 1.3.1 API
    |public class Bag<T> implements Iterable<T>
    |       |Bag()
    |void   |add(T t)
    |boolean|isEmpty()
    |int    |size()
    
    FIFO QUEUE
    |public class Queue<T> implements Iterable<T>
    |       |Queue()
    |T      |enqueue(T t)
    |void   |add(T t)
    |boolean|isEmpty()
    |int    |size() 
    
    LIFO STACK
    |public class Stack<T> implements Iterable<T>
    |       |Stack()
    |void   |push(T t)
    |T t    |pop()
    |boolean|isEmpty()
    |int    |size()
